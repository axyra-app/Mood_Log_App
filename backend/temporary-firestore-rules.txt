# REGLAS TEMPORALES DE FIRESTORE PARA PRUEBAS
# ⚠️ IMPORTANTE: Estas reglas son solo para pruebas. NO las uses en producción.

service cloud.firestore {
  match /databases/{database}/documents {
    // Reglas temporales más permisivas para debugging
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
